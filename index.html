function getFinalOpenedDoors(numDoors) {
  // initialize all closed doors
  let doors = []
  for (let i = 1; i <= numDoors; i++)
    doors.push("closed");

  /*
    note that every closed door is stated as false
    and every opened door is stated as true in doors array
  */

  // initialize pass = 1
  let num_pass = 1;

  /*
    from this, we will toogle the doors.
    the rules is close the door if the door is opened
    and open the doors if the door is closed
  */
  while (num_pass <= numDoors) {
    for (let i = 1; i <= numDoors; i++) {
      if (i % num_pass == 0) {
        if (doors[i-1] == "open")
          doors[i-1] = "closed";
        else
          doors[i-1] = "open";
      }
    }
    
    // console.log(doors);

    num_pass += 1;
  }

  /*
    after toogle-ing the doors, we will show only the the door numbers which is open
  */
  let openedDoors = [];
  for (let i = 0; i < numDoors; i++) {
    if (doors[i] == "open")
      openedDoors.push(i+1);
  }

  // console.log(doors);
  // console.log(openedDoors);
  return openedDoors;
}

function rms(arr) {
  let sq_sum = 0;

  arr.forEach (i => sq_sum += i * i);

  sq_sum /= arr.length;

  return Math.sqrt(sq_sum);
}

function getDPA(num) {
  // Good luck!

  let numstep = 1;
  let DPA_array = [0, 0, 0];

  while (numstep <= num) {    
    let numset = [];

    for (let i = 1; i < numstep; i++) {
      if (numstep % i == 0) {
        numset.push(i);
      }
    }

    let sum = function(arr) {
    return arr.reduce(function(a,b) {
      return a + b
    }, 0);
  };
    
    sum_result = sum(numset);
    // console.log(sum(numset));

    if (sum_result < num)
      DPA_array[0] += 1;
    else if (sum_result === num)
      DPA_array[1] += 1;
    else
      DPA_array[2] += 1;

    // console.log(numset, sum)

    numstep++;
  }

  return DPA_array;
}

function sumsq(array) {
  let sum = 0;

  for (let i = 0; i < array.length; i++)
    sum += array[i] * array[i];

  return sum;
